Sankey Charts

D3 Sankey Chart

  • See : https://github.com/ricklupton/d3-sankey-diagram
  • This component allows you to drill in and zoom on just the element selected.
  • You must include the sankey partials.
  • If you leave the data property empty then the chart will generate sample data.
	
	
	<% include ../partials/sankey %>

	....

  var chartSection = function () {
    var data = {
      nodes: [{ "id": 0, "name": "RuntimeException" },
      { "id": 1, "name": "Login Service" },
      { "id": 2, "name": "DB Service" },
      { "id": 3, "name": "Customer A" },
      { "id": 4, "name": "Customer B" }],
      links: [{ "source": 0, "target": 1, "value": 15 },
      { "source": 0, "target": 2, "value": 5 },
      { "source": 1, "target": 3, "value": 10 },
      { "source": 1, "target": 4, "value": 5 }
      ]
    }

    sankeyChart = new SankeyChart({
      targetId: "chart",
      data: data,
      options: { height: "1000px", width: "1000px" }
    });

    sankeyChart.renderChart(function (selection) {
      alert(selection);
    });
